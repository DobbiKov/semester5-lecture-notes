\section{Mathématiques Générales} 
\subsection{Binomial}
\[
    \binom{n}{k} = \frac{n!}{k!(n-k)!}
\] 
Propriétés utiles:
\begin{enumerate}
    \item $\binom{n}{0} = \binom{n}{n} = 1$
    \item $\binom{n}{1} = n$ 
    \item $\binom{n}{k} = \binom{n-1}{k-1} + \binom{n-1}{k}$
\end{enumerate}
